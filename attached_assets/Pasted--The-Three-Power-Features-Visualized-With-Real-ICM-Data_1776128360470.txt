### What the Heatmap Instantly Reveals

Reading this left to right tells the story of ICM's development:

**The Cloud Pivot (2023→present):** The top row is completely empty for 12 years, then explodes with █ intensity starting at v2024.0. Cloud went from nonexistent to the single largest development investment in one release cycle. This is the most dramatic pattern in the entire dataset.

**The 1D→2D Migration:** In early versions (2011-2014), 1D Hydraulics (🔧) is consistently the densest row. By 2019-2021, 2D Modeling (🌊) has overtaken it. The field moved from pipe networks to surface flooding.

**Rainfall Matured and Stopped:** The 🌧️ row shows steady activity from 2016-2021 (FEH2013, NOAA ATLAS 14, Euler Type II), then goes completely dark. Rainfall generation was "done" — the investment shifted elsewhere.

**SWMM: A Burst Then Maintenance:** The 💧 row shows a single intense spike at v10.5 (2019) when SWMM was integrated, followed by scattered minor touches. It was a one-time architectural addition, not an evolving capability.

**Water Quality Peaked Early:** The 🧪 row was densest in 2015-2016 (sediment modeling, 1D diffusion), then tapered off. This suggests WQ modeling reached sufficient maturity or wasn't market-driven.

```html
function renderHeatmap(containerId) {
  const container = document.getElementById(containerId);
  const catKeys = Object.keys(ICM_CATEGORIES);
  const versions = ICM_HEATMAP_DATA;

  // Find global max
  let maxVal = 0;
  versions.forEach(([,,, cats]) => {
    Object.values(cats).forEach(v => { if (v > maxVal) maxVal = v; });
  });

  // Compute row totals
  const rowTotals = {};
  catKeys.forEach(cat => {
    rowTotals[cat] = versions.reduce((sum, [,,, cats]) => 
      sum + (cats[cat] || 0), 0);
  });

  // Sort categories by total (descending)
  const sortedCats = [...catKeys].sort((a, b) => 
    (rowTotals[b] || 0) - (rowTotals[a] || 0));

  // Column totals
  const colTotals = versions.map(([,,, cats]) => 
    Object.values(cats).reduce((s, v) => s + v, 0));

  // Build table
  let html = `
    <div class="hm-wrap">
      <div class="hm-ctrl">
        <label>Color scheme:
          <select id="hm-scheme">
            <option value="category">Category colors</option>
            <option value="blue">Blue scale</option>
            <option value="heat">Heat (yellow→red)</option>
            <option value="viridis">Viridis</option>
          </select>
        </label>
        <label>
          <input type="checkbox" id="hm-normalize"> 
          Normalize by row
        </label>
        <label>
          <input type="checkbox" id="hm-values" checked> 
          Show values
        </label>
      </div>
      <table class="hm-table">
        <thead><tr>
          <th class="hm-corner">Category / Version →</th>
          ${versions.map(([id,,isMajor]) => `
            <th class="hm-ch ${isMajor ? 'major' : ''}" data-v="${id}">
              <span class="vl">${id}</span>
              ${isMajor ? '<div style="font-size:8px">★</div>' : ''}
            </th>
          `).join('')}
          <th class="hm-ch" style="color:#e2e8f0;font-weight:700">Σ</th>
        </tr></thead>
        <tbody>
          ${sortedCats.map(cat => {
            const meta = ICM_CATEGORIES[cat];
            return `<tr data-cat="${cat}">
              <td class="hm-rh" style="border-left:4px solid ${meta.color}">
                <span class="ri">${meta.icon}</span>
                ${meta.label}
                <span class="rt">${rowTotals[cat]}</span>
              </td>
              ${versions.map(([vId,,, cats]) => {
                const val = cats[cat] || 0;
                const intensity = maxVal > 0 ? val / maxVal : 0;
                const bg = val > 0 
                  ? cellColor(intensity, meta.color) 
                  : 'transparent';
                const featureKey = `${cat}|${vId}`;
                return `<td class="hm-c ${val > 0 ? 'hv' : ''}"
                  data-cat="${cat}" data-v="${vId}" data-val="${val}"
                  data-features="${featureKey}"
                  style="background:${bg}"
                  >${val > 0 ? `<span class="cv">${val}</span>` : ''}</td>`;
              }).join('')}
              <td class="hm-rt" style="color:${meta.color}">${rowTotals[cat]}</td>
            </tr>`;
          }).join('')}
        </tbody>
        <tfoot><tr>
          <td class="hm-corner" style="font-weight:600">Total</td>
          ${colTotals.map((t, i) => `
            <td class="hm-ft">${t}</td>
          `).join('')}
          <td class="hm-ft gt">809</td>
        </tr></tfoot>
      </table>
    </div>
    <div class="hm-tip" id="hm-tip"></div>
  `;

  container.innerHTML = html;

      let refMax = maxVal;
      if (normalize) {
        refMax = 0;
        versions.forEach(([,,, cats]) => {
          if ((cats[cat] || 0) > refMax) refMax = cats[cat];
        });
      }

      const intensity = refMax > 0 ? val / refMax : 0;

      if (scheme === 'category') {
        cell.style.background = cellColor(intensity, meta.color);

        cell.style.background = schemeColor(intensity, scheme);

function cellColor(intensity, hexColor) {
  if (intensity === 0) return 'transparent';
  const r = parseInt(hexColor.slice(1, 3), 16);
  const g = parseInt(hexColor.slice(3, 5), 16);
  const b = parseInt(hexColor.slice(5, 7), 16);
  const a = (0.15 + intensity * 0.8).toFixed(2);
  return `rgba(${r},${g},${b},${a})`;
}

function schemeColor(intensity, scheme) {
  if (intensity === 0) return 'transparent';
  const a = (0.15 + intensity * 0.8).toFixed(2);
  switch (scheme) {
    case 'blue':    return `rgba(59,130,246,${a})`;
    case 'heat': {
      const g = Math.round(255 * (1 - intensity * 0.8));
      return `rgba(255,${g},${Math.round(50*(1-intensity))},${a})`;
    }
    case 'viridis': {
      const t = intensity;
      return `rgba(${Math.round(68+t*185)},${Math.round(1+t*230)},${Math.round(84-t*47)},${a})`;
    }
    default: return `rgba(99,102,241,${a})`;
  }
}
```

The chain detector scans all 809 features looking for three kinds of connections:

```
DETECTION METHOD 1: Explicit Links
Feature A.relatedFeatures includes Feature B's ID
    → Direct parent-child relationship

DETECTION METHOD 2: Title Similarity (Jaccard > 0.55)
"Cloud Capabilities Enhancements" ↔ "Cloud Capabilities"
    → Same capability, different versions

DETECTION METHOD 3: Shared Rare Keywords  
Features sharing "subgrid" + "meshing" across versions
    → Topical evolution thread
```

### Comparison Engine — Core Logic

```javascript
// version-compare-engine.js
class VersionComparator {
  constructor(features, versions, categories) {
    this.features = features;
    this.versions = versions;
    this.categories = categories;
  }

  compare(fromId, toId) {
    const from = this.versions.find(v => v.id === fromId);
    const to = this.versions.find(v => v.id === toId);
    if (!from || !to) return null;

    // Ensure from < to
    const [older, newer] = from.seq <= to.seq ? [from, to] : [to, from];

    // All features added between these versions
    const newFeatures = this.features.filter(f => {
      const fv = this.versions.find(v => v.id === f.version);
      return fv && fv.seq > older.seq && fv.seq <= newer.seq;
    });

    // Versions spanned
    const spanned = this.versions
      .filter(v => v.seq > older.seq && v.seq <= newer.seq)
      .sort((a, b) => a.seq - b.seq);

    // Group by category
    const byCategory = {};
    newFeatures.forEach(f => {
      f.categories.forEach(cat => {
        if (!byCategory[cat]) byCategory[cat] = [];
        byCategory[cat].push(f);
      });
    });

    // Group by version
    const byVersion = {};
    newFeatures.forEach(f => {
      if (!byVersion[f.version]) byVersion[f.version] = [];
      byVersion[f.version].push(f);
    });

    // Category shift analysis
    const categoryShift = this.analyzeCategoryShift(older, newer);

    // Detect evolution chains in this range
    const chains = this.detectChainsInRange(newFeatures, older.seq, newer.seq);

    return {
      from: older,
      to: newer,
      newFeatures,
      spanned,
      byCategory,
      byVersion,
      chains,
      categoryShift,
      stats: {
        total: newFeatures.length,
        majorReleases: spanned.filter(v => v.isMajor).length,
        versionsCount: spanned.length,
        topCategories: Object.entries(byCategory)
          .map(([cat, feats]) => ({
            cat, 
            ...this.categories[cat], 
            count: feats.length
          }))
          .sort((a, b) => b.count - a.count)
          .slice(0, 5)
      }
    };
  }

  analyzeCategoryShift(older, newer) {
    // Compare category proportions near version A vs near version B
    const windowSize = 3;

    const nearOlder = this.features.filter(f => {
      const fv = this.versions.find(v => v.id === f.version);
      return fv && Math.abs(fv.seq - older.seq) <= windowSize;
    });

    const nearNewer = this.features.filter(f => {
      const fv = this.versions.find(v => v.id === f.version);
      return fv && Math.abs(fv.seq - newer.seq) <= windowSize;
    });

    const distOlder = this.categoryProportions(nearOlder);
    const distNewer = this.categoryProportions(nearNewer);

    return Object.keys(this.categories)
      .map(cat => ({
        cat,
        ...this.categories[cat],
        before: distOlder[cat] || 0,
        after: distNewer[cat] || 0,
        shift: (distNewer[cat] || 0) - (distOlder[cat] || 0)
      }))
      .filter(c => Math.abs(c.shift) > 0.02)
      .sort((a, b) => Math.abs(b.shift) - Math.abs(a.shift));
  }

  categoryProportions(features) {
    const counts = {};
    let total = 0;
    features.forEach(f => {
      f.categories.forEach(cat => {
        counts[cat] = (counts[cat] || 0) + 1;
        total++;
      });
    });
    Object.keys(counts).forEach(k => { counts[k] /= total || 1; });
    return counts;
  }

  detectChainsInRange(features, fromSeq, toSeq) {
    // Group features by title similarity
    const chains = [];
    const used = new Set();

    features.forEach(anchor => {
      if (used.has(anchor.id)) return;

      const cluster = [anchor];
      used.add(anchor.id);

      features.forEach(candidate => {
        if (used.has(candidate.id)) return;
        if (candidate.version === anchor.version) return;
        if (titleSimilarity(anchor.title, candidate.title) >= 0.55) {
          cluster.push(candidate);
          used.add(candidate.id);
        }
      });

      if (cluster.length >= 2) {
        chains.push({
          title: cluster[0].title.replace(/\s*(Enhancement|Continued|Improvement)s?\s*$/i, ''),
          features: cluster.sort((a, b) => {
            const va = this.versions.find(v => v.id === a.version)?.seq || 0;
            const vb = this.versions.find(v => v.id === b.version)?.seq || 0;
            return va - vb;
          }),
          category: cluster[0].categories[0]
        });
      }
    });

    return chains.sort((a, b) => b.features.length - a.features.length);
  }
}

function titleSimilarity(a, b) {
  const stop = new Set(['the','a','an','and','or','in','on','to','for',
    'of','with','by','new','now','also','features','improvements',
    'enhancements','continued','support']);
  const tokenize = s => s.toLowerCase().split(/[^a-z0-9]+/)
    .filter(w => w.length > 2 && !stop.has(w));
  const wA = new Set(tokenize(a));
  const wB = new Set(tokenize(b));
  if (!wA.size || !wB.size) return 0;
  const inter = [...wA].filter(w => wB.has(w)).length;
  return inter / new Set([...wA, ...wB]).size;
}
```

```javascript
// event-bus.js — the glue connecting all three features
class ICMEventBus {
  constructor() {
    this.listeners = {};
  }

  on(event, callback) {
    if (!this.listeners[event]) this.listeners[event] = [];
    this.listeners[event].push(callback);
  }

  emit(event, data) {
    (this.listeners[event] || []).forEach(cb => cb(data));
  }
}

// Initialize
const bus = new ICMEventBus();

// ─── HEATMAP emits ───
bus.on('heatmap:cell-click', ({ category, version }) => {
  // Tell timeline to filter
  bus.emit('timeline:filter', { categories: [category], version });
  
  // Tell chains to highlight
  bus.emit('chains:highlight-category', { category, version });
  
  // Tell comparison to suggest
  bus.emit('comparison:suggest-from', { version });
});

bus.on('heatmap:row-click', ({ category }) => {
  // Show all chains for this category
  bus.emit('chains:filter-category', { category });
  
  // Filter timeline to this category
  bus.emit('timeline:filter', { categories: [category] });
});

// ─── CHAINS emits ───
bus.on('chains:node-click', ({ featureId, version, category }) => {
  // Scroll timeline to feature
  bus.emit('timeline:scroll-to', { featureId });
  
  // Highlight heatmap cell
  bus.emit('heatmap:highlight-cell', { category, version });
});

bus.on('chains:range-select', ({ fromVersion, toVersion }) => {
  // Pre-fill comparison
  bus.emit('comparison:set-range', { fromVersion, toVersion });
});

// ─── COMPARISON emits ───
bus.on('comparison:complete', ({ fromVersion, toVersion, result }) => {
  // Overlay on heatmap
  bus.emit('heatmap:show-range', { fromVersion, toVersion });
  
  // Highlight chains in range
  bus.emit('chains:show-in-range', { fromVersion, toVersion });
  
  // Mark timeline features
  bus.emit('timeline:mark-new-since', { version: fromVersion });
});

bus.on('comparison:feature-click', ({ featureId }) => {
  // Scroll timeline
  bus.emit('timeline:scroll-to', { featureId });
});

// ─── TIMELINE responds ───
bus.on('timeline:filter', ({ categories, version }) => {
  app.filters.state.categories = categories || [];
  if (version) {
    // Scroll to that version section
    document.getElementById(`version-${version}`)
      ?.scrollIntoView({ behavior: 'smooth' });
  }
  app.filters.applyAndRender(
    document.getElementById('filter-container')
  );
});

bus.on('timeline:scroll-to', ({ featureId }) => {
  const el = document.getElementById(featureId);
  if (!el) return;
  el.scrollIntoView({ behavior: 'smooth', block: 'center' });
  el.classList.add('pulse-highlight');
  setTimeout(() => el.classList.remove('pulse-highlight'), 2500);
});

bus.on('timeline:mark-new-since', ({ version }) => {
  app.baseline.setBaseline(version);
});

// ─── HEATMAP responds ───
bus.on('heatmap:highlight-cell', ({ category, version }) => {
  // Remove previous highlights
  document.querySelectorAll('.hm-c.highlighted')
    .forEach(c => c.classList.remove('highlighted'));
  
  // Add highlight
  const cell = document.querySelector(
    `.hm-c[data-cat="${category}"][data-v="${version}"]`
  );
  if (cell) {
    cell.classList.add('highlighted');
    cell.scrollIntoView({ behavior: 'smooth', block: 'center' });
  }
});

bus.on('heatmap:show-range', ({ fromVersion, toVersion }) => {
  const fromSeq = app.versions.find(v => v.id === fromVersion)?.seq || 0;
  const toSeq = app.versions.find(v => v.id === toVersion)?.seq || 999;
  
  document.querySelectorAll('.hm-c').forEach(cell => {
    const vId = cell.dataset.v;
    const vSeq = app.versions.find(v => v.id === vId)?.seq || 0;
    
    if (vSeq > fromSeq && vSeq <= toSeq) {
      cell.classList.add('in-comparison-range');
    } else {
      cell.classList.add('outside-comparison-range');
    }
  });
});

// ─── CHAINS responds ───
bus.on('chains:highlight-category', ({ category, version }) => {
  document.querySelectorAll('.chain-card').forEach(card => {
    if (card.dataset.category === category) {
      card.classList.add('chain-highlighted');
      // If version is in this chain, expand and scroll to that node
      const chain = app.chains.find(c => c.id === card.dataset.chainId);
      if (chain?.versions.includes(version)) {
        card.open = true;
      }
    } else {
      card.classList.add('chain-dimmed');
    }
  });
});

bus.on('chains:filter-category', ({ category }) => {
  document.querySelectorAll('.chain-card').forEach(card => {
    card.style.display = 
      card.dataset.category === category ? '' : 'none';
  });
});
```

### Summary: The Three Features as a System

| Feature | Role | Answers | Feeds Into |
|---|---|---|---|
| **Heatmap** | Discovery | *Where are the hot spots?* | Chains (explain patterns), Comparison (set range), Timeline (filter) |
| **Chains** | Narrative | *How did this capability evolve?* | Timeline (scroll to feature), Heatmap (highlight cell), Comparison (suggest range) |
| **Comparison** | Decision | *What do I gain by upgrading?* | Timeline (mark new features), Heatmap (overlay range), Chains (show bridges), Export (action) |

The heatmap finds the pattern. The chains explain the pattern. The comparison acts on the pattern. Each feature makes the other two more powerful — and all three feed back into the timeline through scroll-to, filter, and highlight interactions.
