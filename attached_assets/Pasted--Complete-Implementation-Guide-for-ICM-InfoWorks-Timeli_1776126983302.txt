### Full-Text Search Engine (Client-Side)

For 809 features, client-side search is fast enough. Use a lightweight index:

```javascript
// search-engine.js
class FeatureSearchEngine {
  constructor(features) {
    this.features = features;
    this.index = this.buildIndex(features);
  }

  buildIndex(features) {
    // Build inverted index: word → [feature IDs]
    const index = new Map();
    
    features.forEach(feature => {
      const text = [
        feature.title,
        feature.description,
        ...feature.keywords,
        feature.version
      ].join(' ').toLowerCase();
      
      // Tokenize: split on non-alphanumeric, filter short words
      const words = text
        .split(/[^a-z0-9]+/)
        .filter(w => w.length > 1);
      
      // Also generate bigrams for phrase-ish matching
      const bigrams = [];
      for (let i = 0; i < words.length - 1; i++) {
        bigrams.push(words[i] + ' ' + words[i + 1]);
      }
      
      [...new Set([...words, ...bigrams])].forEach(word => {
        if (!index.has(word)) index.set(word, new Set());
        index.get(word).add(feature.id);
      });
    });
    
    return index;
  }

  search(query) {
    if (!query || query.trim().length < 2) return [];
    
    const queryTerms = query.toLowerCase()
      .split(/[^a-z0-9]+/)
      .filter(w => w.length > 1);
    
    if (queryTerms.length === 0) return [];

    // Score each feature by how many query terms it matches
    const scores = new Map();
    
    queryTerms.forEach(term => {
      // Exact match
      this.index.forEach((featureIds, indexedWord) => {
        if (indexedWord === term) {
          featureIds.forEach(id => {
            scores.set(id, (scores.get(id) || 0) + 10);
          });
        }
        // Prefix match (e.g., "sub" matches "subgrid")
        else if (indexedWord.startsWith(term) && term.length >= 3) {
          featureIds.forEach(id => {
            scores.set(id, (scores.get(id) || 0) + 5);
          });
        }
        // Contains match
        else if (indexedWord.includes(term) && term.length >= 4) {
          featureIds.forEach(id => {
            scores.set(id, (scores.get(id) || 0) + 2);
          });
        }
      });
    });

    // Title matches get a boost
    this.features.forEach(f => {
      if (scores.has(f.id)) {
        const titleLower = f.title.toLowerCase();
        queryTerms.forEach(term => {
          if (titleLower.includes(term)) {
            scores.set(f.id, scores.get(f.id) + 15);
          }
        });
      }
    });

    return [...scores.entries()]
      .sort((a, b) => b[1] - a[1])
      .map(([id, score]) => ({
        feature: this.features.find(f => f.id === id),
        score
      }));
  }
}
```

### Search-Result Highlighting Using CSS Custom Highlight API

This is the modern approach — it doesn't modify the DOM, so it works safely with any framework:

```javascript
// highlight.js — uses the CSS Custom Highlight API
// (Chrome 105+, Edge 105+, Safari 17.2+, Firefox 132+)

class SearchHighlighter {
  constructor() {
    this.highlightName = 'search-results';
  }

  highlight(query, containerSelector = '.feature-card') {
    // Clear previous highlights
    CSS.highlights?.delete(this.highlightName);
    
    if (!query || query.trim().length < 2) return;

    const terms = query.toLowerCase()
      .split(/[^a-z0-9]+/)
      .filter(w => w.length > 1);

    const ranges = [];
    const containers = document.querySelectorAll(containerSelector);

    containers.forEach(container => {
      // Walk all text nodes in the container
      const walker = document.createTreeWalker(
        container,
        NodeFilter.SHOW_TEXT,
        null
      );

      while (walker.nextNode()) {
        const textNode = walker.currentNode;
        const text = textNode.textContent.toLowerCase();

        terms.forEach(term => {
          let startPos = 0;
          while (startPos < text.length) {
            const index = text.indexOf(term, startPos);
            if (index === -1) break;

            const range = new Range();
            range.setStart(textNode, index);
            range.setEnd(textNode, index + term.length);
            ranges.push(range);
            startPos = index + term.length;
          }
        });
      }
    });

    if (ranges.length > 0 && CSS.highlights) {
      const highlight = new Highlight(...ranges);
      CSS.highlights.set(this.highlightName, highlight);
    }

    return ranges.length;
  }

  clear() {
    CSS.highlights?.delete(this.highlightName);
  }
}
```

```javascript
// highlight-fallback.js
function highlightFallback(query, containerSelector) {
  // Remove existing marks
  document.querySelectorAll('mark.search-highlight')
    .forEach(el => {
      el.replaceWith(el.textContent);
    });
  
  if (!query || query.trim().length < 2) return;

  const terms = query.toLowerCase()
    .split(/[^a-z0-9]+/)
    .filter(w => w.length > 1);
  
  // Build regex that matches any term
  const escapedTerms = terms.map(t => 
    t.replace(/[.*+?^${}()|[\]\\]/g, '\\$&')
  );
  const regex = new RegExp(`(${escapedTerms.join('|')})`, 'gi');

  document.querySelectorAll(containerSelector).forEach(container => {
    const walker = document.createTreeWalker(
      container, NodeFilter.SHOW_TEXT, null
    );
    const textNodes = [];
    while (walker.nextNode()) textNodes.push(walker.currentNode);

    textNodes.forEach(node => {
      if (!regex.test(node.textContent)) return;
      regex.lastIndex = 0;
      
      const fragment = document.createDocumentFragment();
      let lastIndex = 0;
      let match;

      while ((match = regex.exec(node.textContent)) !== null) {
        // Text before match
        if (match.index > lastIndex) {
          fragment.appendChild(
            document.createTextNode(
              node.textContent.slice(lastIndex, match.index)
            )
          );
        }
        // The highlighted match
        const mark = document.createElement('mark');
        mark.className = 'search-highlight';
        mark.textContent = match[0];
        fragment.appendChild(mark);
        lastIndex = regex.lastIndex;
      }
      // Text after last match
      if (lastIndex < node.textContent.length) {
        fragment.appendChild(
          document.createTextNode(node.textContent.slice(lastIndex))
        );
      }
      node.parentNode.replaceChild(fragment, node);
    });
  });
}
```

### "My Version" Baseline Selector

```javascript
// baseline-selector.js
class BaselineSelector {
  constructor(versions, features, onBaselineChange) {
    this.versions = versions;
    this.features = features;
    this.onBaselineChange = onBaselineChange;
    this.baseline = this.loadBaseline();
  }

  loadBaseline() {
    return localStorage.getItem('icm-my-version') || null;
  }

  setBaseline(versionId) {
    this.baseline = versionId;
    localStorage.setItem('icm-my-version', versionId);
    this.onBaselineChange(versionId);
  }

  clearBaseline() {
    this.baseline = null;
    localStorage.removeItem('icm-my-version');
    this.onBaselineChange(null);
  }

  getNewFeatures() {
    if (!this.baseline) return [];
    const baselineVersion = this.versions.find(v => v.id === this.baseline);
    if (!baselineVersion) return [];
    
    return this.features.filter(f => {
      const fVersion = this.versions.find(v => v.id === f.version);
      return fVersion && fVersion.seq > baselineVersion.seq;
    });
  }

  getUpgradeSummary() {
    const newFeatures = this.getNewFeatures();
    if (newFeatures.length === 0) return null;

    // Count by category
    const byCat = {};
    newFeatures.forEach(f => {
      f.categories.forEach(cat => {
        byCat[cat] = (byCat[cat] || 0) + 1;
      });
    });

    // Count by impact
    const byImpact = { major: 0, minor: 0, maintenance: 0 };
    newFeatures.forEach(f => {
      byImpact[f.impactLevel] = (byImpact[f.impactLevel] || 0) + 1;
    });

    // Versions spanned
    const versionIds = [...new Set(newFeatures.map(f => f.version))];
    
    return {
      totalFeatures: newFeatures.length,
      versionsSpanned: versionIds.length,
      byCategory: Object.entries(byCat)
        .sort((a, b) => b[1] - a[1]),
      byImpact,
      features: newFeatures
    };
  }

  renderSelector(containerEl) {
    containerEl.innerHTML = `
      <div class="baseline-selector">
        <label for="my-version">My Current Version:</label>
        <select id="my-version">
          <option value="">— Show all versions —</option>
          ${this.versions.map(v => `
            <option value="${v.id}" 
              ${v.id === this.baseline ? 'selected' : ''}
              ${v.isMajor ? 'class="major-version"' : ''}>
              ${v.id} ${v.isMajor ? '★' : ''} (${v.date})
            </option>
          `).join('')}
        </select>
        ${this.baseline ? `
          <div class="baseline-summary" id="baseline-summary">
            ${this.renderSummary()}
          </div>
        ` : ''}
      </div>
    `;

    containerEl.querySelector('#my-version')
      .addEventListener('change', (e) => {
        if (e.target.value) {
          this.setBaseline(e.target.value);
        } else {
          this.clearBaseline();
        }
        this.renderSelector(containerEl); // re-render
      });
  }

  renderSummary() {
    const summary = this.getUpgradeSummary();
    if (!summary) return '<p>You are on the latest version.</p>';

    return `
      <div class="upgrade-banner">
        <strong>${summary.totalFeatures} new features</strong> 
        across ${summary.versionsSpanned} versions since ${this.baseline}
        <div class="impact-pills">
          ${summary.byImpact.major > 0 ? 
            `<span class="pill major">${summary.byImpact.major} major</span>` : ''}
          ${summary.byImpact.minor > 0 ? 
            `<span class="pill minor">${summary.byImpact.minor} minor</span>` : ''}
          ${summary.byImpact.maintenance > 0 ? 
            `<span class="pill maint">${summary.byImpact.maintenance} maint</span>` : ''}
        </div>
        <div class="category-breakdown">
          ${summary.byCategory.slice(0, 5).map(([cat, count]) => `
            <span class="cat-chip" style="border-color: ${CATEGORIES[cat]?.color || '#999'}">
              ${CATEGORIES[cat]?.icon || '•'} ${CATEGORIES[cat]?.label || cat}: ${count}
            </span>
          `).join('')}
        </div>
        <button id="export-upgrade-report">Export Upgrade Report</button>
      </div>
    `;
  }
}
```

### Wiring Search + Baseline Together

```javascript
// app.js — main controller
class TimelineApp {
  constructor() {
    this.search = new FeatureSearchEngine(FEATURES_DB);
    this.highlighter = new SearchHighlighter();
    this.baseline = new BaselineSelector(
      VERSIONS, FEATURES_DB, 
      (versionId) => this.onBaselineChange(versionId)
    );
    
    this.activeFilters = {
      query: '',
      categories: [],
      baselineVersion: this.baseline.baseline
    };

    this.init();
  }

  init() {
    // Render baseline selector
    this.baseline.renderSelector(
      document.getElementById('baseline-container')
    );

    // Search input with debounce
    const searchInput = document.getElementById('feature-search');
    let debounceTimer;
    searchInput.addEventListener('input', (e) => {
      clearTimeout(debounceTimer);
      debounceTimer = setTimeout(() => {
        this.activeFilters.query = e.target.value;
        this.applyFilters();
      }, 200);
    });

    // Initial render
    this.renderTimeline(FEATURES_DB);
  }

  applyFilters() {
    let results = FEATURES_DB;

    // Text search
    if (this.activeFilters.query.length >= 2) {
      const searchResults = this.search.search(this.activeFilters.query);
      const matchedIds = new Set(searchResults.map(r => r.feature.id));
      results = results.filter(f => matchedIds.has(f.id));
    }

    // Category filter
    if (this.activeFilters.categories.length > 0) {
      results = results.filter(f =>
        f.categories.some(c => this.activeFilters.categories.includes(c))
      );
    }

    this.renderTimeline(results);

    // Apply highlighting after DOM update
    requestAnimationFrame(() => {
      this.highlighter.highlight(
        this.activeFilters.query, 
        '.feature-card'
      );
    });

    // Update result count
    document.getElementById('result-count').textContent = 
      this.activeFilters.query 
        ? `${results.length} of ${FEATURES_DB.length} features` 
        : `${FEATURES_DB.length} features`;
  }

  onBaselineChange(versionId) {
    this.activeFilters.baselineVersion = versionId;
    this.applyFilters();
  }

  renderTimeline(features) {
    // Group by version
    const grouped = new Map();
    features.forEach(f => {
      if (!grouped.has(f.version)) grouped.set(f.version, []);
      grouped.get(f.version).push(f);
    });

    const container = document.getElementById('timeline');
    const baselineSeq = this.baseline.baseline 
      ? VERSIONS.find(v => v.id === this.baseline.baseline)?.seq || 0 
      : 0;

    container.innerHTML = [...grouped.entries()]
      .sort((a, b) => {
        const va = VERSIONS.find(v => v.id === a[0]);
        const vb = VERSIONS.find(v => v.id === b[0]);
        return (vb?.seq || 0) - (va?.seq || 0);
      })
      .map(([versionId, vFeatures]) => {
        const vMeta = VERSIONS.find(v => v.id === versionId);
        const isNew = vMeta && vMeta.seq > baselineSeq && baselineSeq > 0;

        return `
          <section class="version-section ${isNew ? 'new-since-baseline' : ''}" 
                   id="version-${versionId}">
            <h2>
              Version ${versionId}
              ${vMeta?.isMajor ? '<span class="major-badge">Major Release</span>' : ''}
              ${isNew ? '<span class="new-badge">NEW</span>' : ''}
            </h2>
            <time>${vMeta?.date || ''}</time>
            <span class="feature-count">${vFeatures.length} features</span>
            
            ${vFeatures.map(f => `
              <article class="feature-card ${isNew ? 'new-since-baseline' : ''}"
                       id="${f.id}"
                       data-categories="${f.categories.join(',')}"
                       data-impact="${f.impactLevel}">
                <h3>
                  <a href="#${f.id}" class="feature-anchor">#</a>
                  ${f.title}
                  <button class="ask-ai-btn" 
                    data-feature-id="${f.id}" 
                    title="Ask AI about this feature">💬</button>
                </h3>
                <p>${f.description}</p>
                <div class="feature-tags">
                  ${f.categories.map(c => `
                    <span class="tag" 
                      style="background: ${CATEGORIES[c]?.color}22; 
                             color: ${CATEGORIES[c]?.color}; 
                             border: 1px solid ${CATEGORIES[c]?.color}44"
                      data-category="${c}">
                      ${CATEGORIES[c]?.icon} ${CATEGORIES[c]?.label}
                    </span>
                  `).join('')}
                </div>
              </article>
            `).join('')}
          </section>
        `;
      }).join('');
  }
}
```

## 2. Making AI Chat Verifiable with Citations and Feature Links

This is the highest-trust-impact change. The principle: **every AI claim about a feature should link to the specific feature card on the page.**

### Architecture: Retrieval-Augmented Generation (RAG) Over Your Feature Database

```javascript
// ai-chat.js
class VerifiableAIChat {
  constructor(featuresDB, searchEngine) {
    this.features = featuresDB;
    this.search = searchEngine;
    this.conversationHistory = [];
  }

  // Step 1: Retrieve relevant features BEFORE calling the LLM
  retrieveContext(userQuery, maxResults = 15) {
    const results = this.search.search(userQuery);
    return results.slice(0, maxResults).map(r => r.feature);
  }

  // Step 2: Build system prompt with retrieved features as context
  buildPrompt(userQuery, retrievedFeatures) {
    const contextBlock = retrievedFeatures.map((f, i) => `
[FEATURE-${i + 1}]
ID: ${f.id}
Version: ${f.version} (${f.versionDate})
Title: ${f.title}
Description: ${f.description}
Categories: ${f.categories.join(', ')}
`).join('\n---\n');

    return {
      system: `You are an expert assistant for Autodesk InfoWorks ICM, 
a hydraulic and hydrological modeling platform. You help users understand 
features, plan upgrades, and compare versions.

CRITICAL RULES:
1. ONLY reference features provided in the CONTEXT block below.
2. For EVERY claim about a feature, include a citation in this exact format: 
   [ref:FEATURE-ID](feature-id-here)
   Example: [ref:v2027.0-network-design](v2027.0-network-design)
3. If you don't have enough context to answer, say so explicitly.
4. NEVER fabricate features, version numbers, or capabilities.
5. When comparing versions, cite features from BOTH versions.

CONTEXT (retrieved features relevant to this query):
${contextBlock}

The user's current baseline version (if set): ${this.baselineVersion || 'not set'}`,
      
      user: userQuery
    };
  }

  // Step 3: Call LLM API (adapt to whichever provider you use)
  async chat(userQuery) {
    // Retrieve relevant features
    const context = this.retrieveContext(userQuery);
    
    // Build grounded prompt
    const { system, user } = this.buildPrompt(userQuery, context);
    
    // Call your LLM endpoint
    const response = await fetch('/api/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: this.selectedModel,  // claude, gpt, gemini, deepseek
        messages: [
          { role: 'system', content: system },
          ...this.conversationHistory,
          { role: 'user', content: user }
        ],
        temperature: 0.3  // lower = more faithful to context
      })
    });

    const data = await response.json();
    const aiMessage = data.choices[0].message.content;

    // Store in history
    this.conversationHistory.push(
      { role: 'user', content: user },
      { role: 'assistant', content: aiMessage }
    );

    // Post-process: convert citation markers to clickable links
    const processedMessage = this.processCitations(aiMessage, context);
    
    return processedMessage;
  }

  // Step 4: Convert [ref:...] markers into interactive citations
  processCitations(text, contextFeatures) {
    const citationRegex = /\[ref:([\w\d\.\-]+)\]\(([\w\d\.\-]+)\)/g;
    const citations = [];
    
    const processedHtml = text.replace(citationRegex, (match, label, featureId) => {
      const feature = contextFeatures.find(f => f.id === featureId);
      if (!feature) {
        // Citation references a feature not in context — flag it
        return `<span class="citation citation-unverified" 
                  title="⚠️ Could not verify this reference">${label}</span>`;
      }

      citations.push(feature);
      const citNum = citations.length;
      
      return `<a href="#${feature.id}" 
                 class="citation citation-verified"
                 data-feature-id="${feature.id}"
                 data-version="${feature.version}"
                 title="${feature.title} (v${feature.version})"
                 onclick="app.scrollToFeature('${feature.id}')">
                [${citNum}]
              </a>`;
    });

    return {
      html: processedHtml,
      citations: citations,
      citationCount: citations.length,
      hasUnverified: processedHtml.includes('citation-unverified')
    };
  }
}
```

```javascript
// filters.js
class FilterSystem {
  constructor(features, categories, versions, onFilterChange) {
    this.features = features;
    this.categories = categories;
    this.versions = versions;
    this.onFilterChange = onFilterChange;
    
    this.state = {
      query: '',
      categories: [],          // active category filters
      impactLevels: [],        // ['major', 'minor', 'maintenance']
      versionRange: [null, null], // [from, to]
      onlyMajorReleases: false,
      onlyNewSinceBaseline: false
    };
  }

  render(containerEl) {
    containerEl.innerHTML = `
      <div class="filter-bar">
        <!-- Search -->
        <div class="filter-group">
          <input type="search" id="feature-search" 
                 placeholder="Search features..."
                 value="${this.state.query}">
        </div>

        <!-- Category chips -->
        <div class="filter-group">
          <label>Categories</label>
          <div class="chip-container" id="category-chips">
            ${Object.entries(this.categories).map(([key, meta]) => {
              const count = this.features.filter(f => 
                f.categories.includes(key)
              ).length;
              const isActive = this.state.categories.includes(key);
              return `
                <button class="filter-chip ${isActive ? 'active' : ''}"
                        data-category="${key}"
                        style="${isActive ? 
                          `background: ${meta.color}33; border-color: ${meta.color}` : 
                          ''}">
                  ${meta.icon} ${meta.label} 
                  <span class="chip-count">${count}</span>
                </button>`;
            }).join('')}
          </div>
        </div>

        <!-- Impact level toggles -->
        <div class="filter-group">
          <label>Impact</label>
          <div class="toggle-group">
            ${['major', 'minor', 'maintenance'].map(level => `
              <button class="filter-toggle ${this.state.impactLevels.includes(level) ? 'active' : ''}"
                      data-impact="${level}">
                ${level === 'major' ? '🔴' : level === 'minor' ? '🔵' : '⚪'} 
                ${level}
              </button>
            `).join('')}
          </div>
        </div>

        <!-- Version range slider -->
        <div class="filter-group">
          <label>Version Range</label>
          <div class="version-range">
            <select id="version-from">
              <option value="">Earliest</option>
              ${this.versions.sort((a,b) => a.seq - b.seq).map(v => `
                <option value="${v.id}">${v.id}</option>
              `).join('')}
            </select>
            <span>→</span>
            <select id="version-to">
              <option value="">Latest</option>
              ${this.versions.sort((a,b) => b.seq - a.seq).map(v => `
                <option value="${v.id}">${v.id}</option>
              `).join('')}
            </select>
          </div>
        </div>

        <!-- Quick filters -->
        <div class="filter-group">
          <label>Quick Filters</label>
          <button class="quick-filter" data-action="major-only">
            Major releases only
          </button>
          <button class="quick-filter" data-action="last-year">
            Last 12 months
          </button>
          <button class="quick-filter" data-action="2d-cloud">
            2D + Cloud features
          </button>
        </div>

        <!-- Active filter summary + clear -->
        <div class="filter-summary" id="filter-summary"></div>
      </div>
    `;

    this.bindEvents(containerEl);
  }

  bindEvents(containerEl) {
    // Category chips
    containerEl.querySelectorAll('.filter-chip').forEach(chip => {
      chip.addEventListener('click', () => {
        const cat = chip.dataset.category;
        if (this.state.categories.includes(cat)) {
          this.state.categories = this.state.categories.filter(c => c !== cat);
        } else {
          this.state.categories.push(cat);
        }
        this.applyAndRender(containerEl);
      });
    });

    // Impact toggles
    containerEl.querySelectorAll('.filter-toggle').forEach(btn => {
      btn.addEventListener('click', () => {
        const impact = btn.dataset.impact;
        if (this.state.impactLevels.includes(impact)) {
          this.state.impactLevels = this.state.impactLevels.filter(i => i !== impact);
        } else {
          this.state.impactLevels.push(impact);
        }
        this.applyAndRender(containerEl);
      });
    });

    // Search
    let timer;
    containerEl.querySelector('#feature-search')
      .addEventListener('input', (e) => {
        clearTimeout(timer);
        timer = setTimeout(() => {
          this.state.query = e.target.value;
          this.applyAndRender(containerEl);
        }, 200);
      });

    // Quick filters
    containerEl.querySelectorAll('.quick-filter').forEach(btn => {
      btn.addEventListener('click', () => {
        switch (btn.dataset.action) {
          case 'major-only':
            this.state.onlyMajorReleases = !this.state.onlyMajorReleases;
            break;
          case 'last-year':
            const oneYearAgo = new Date();
            oneYearAgo.setFullYear(oneYearAgo.getFullYear() - 1);
            // find nearest version
            this.state.versionRange[0] = this.versions
              .filter(v => new Date(v.date) >= oneYearAgo)
              .sort((a, b) => a.seq - b.seq)[0]?.id || null;
            break;
          case '2d-cloud':
            this.state.categories = ['2d-modeling', 'cloud'];
            break;
        }
        this.applyAndRender(containerEl);
      });
    });
  }

  getFilteredFeatures() {
    let results = [...this.features];

    // Category filter (OR logic: match any selected category)
    if (this.state.categories.length > 0) {
      results = results.filter(f =>
        f.categories.some(c => this.state.categories.includes(c))
      );
    }

    // Impact filter
    if (this.state.impactLevels.length > 0) {
      results = results.filter(f =>
        this.state.impactLevels.includes(f.impactLevel)
      );
    }

    // Version range
    if (this.state.versionRange[0] || this.state.versionRange[1]) {
      const fromSeq = this.state.versionRange[0]
        ? this.versions.find(v => v.id === this.state.versionRange[0])?.seq || 0
        : 0;
      const toSeq = this.state.versionRange[1]
        ? this.versions.find(v => v.id === this.state.versionRange[1])?.seq || 999
        : 999;
      
      results = results.filter(f => {
        const fSeq = this.versions.find(v => v.id === f.version)?.seq || 0;
        return fSeq >= fromSeq && fSeq <= toSeq;
      });
    }

    // Major releases only
    if (this.state.onlyMajorReleases) {
      const majorVersions = new Set(
        this.versions.filter(v => v.isMajor).map(v => v.id)
      );
      results = results.filter(f => majorVersions.has(f.version));
    }

    return results;
  }

  applyAndRender(containerEl) {
    const filtered = this.getFilteredFeatures();
    this.render(containerEl);
    this.onFilterChange(filtered, this.state);
    this.updateSummary(containerEl, filtered);
  }

  updateSummary(containerEl, filtered) {
    const summaryEl = containerEl.querySelector('#filter-summary');
    const activeCount = this.state.categories.length + 
                        this.state.impactLevels.length +
                        (this.state.query ? 1 : 0) +
                        (this.state.onlyMajorReleases ? 1 : 0);
    
    if (activeCount === 0) {
      summaryEl.innerHTML = '';
      return;
    }

    summaryEl.innerHTML = `
      <span>${filtered.length} of ${this.features.length} features</span>
      <button class="clear-all-filters" onclick="app.filters.clearAll()">
        ✕ Clear all filters
      </button>
    `;
  }

  clearAll() {
    this.state = {
      query: '', categories: [], impactLevels: [],
      versionRange: [null, null],
      onlyMajorReleases: false,
      onlyNewSinceBaseline: false
    };
    const container = document.getElementById('filter-container');
    this.applyAndRender(container);
  }
}
```
